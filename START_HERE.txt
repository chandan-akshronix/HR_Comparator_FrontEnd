  const handleProcessResumes = async () => {
    setIsProcessing(true);
    
    try {
      // Get files that are pending
      const filesToProcess = uploadedResumes
        .filter(r => r.status === 'pending')
        .map(r => r.file);
      
      // Upload to backend (it will parse and store in MongoDB)
      const results = await processResumes(filesToProcess);
      
      // Update UI with results
      setUploadedResumes(prev => prev.map(resume => {
        const result = results.find(r => r.filename === resume.file.name);
        if (result?.success) {
          return {
            ...resume,
            status: 'completed',
            extractedData: {
              name: result.filename,
              skills: ['Parsed from PDF'],
              experience: 'See backend'
            }
          };
        }
        return resume;
      }));
      
      toast.success('All resumes processed and stored in MongoDB!');
    } catch (err) {
      toast.error('Error: ' + err.message);
    } finally {
      setIsProcessing(false);
    }
  };
